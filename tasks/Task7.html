JavaScript Arithmetic & Assignment Operations

// Arithmetic Operations
let a = 10 + 5;
console.log("Addition (10 + 5):", a);

let b = 20 - 7;
console.log("Subtraction (20 - 7):", b);

let c = 4 * 6;
console.log("Multiplication (4 * 6):", c);

let d = 25 / 5;
console.log("Division (25 / 5):", d);

let e = 17 % 4;
console.log("Modulus (17 % 4):", e);

// Assignment Operations
let x = 10;
console.log("Assignment (x = 10):", x);

x += 5;
console.log("Addition Assignment (x += 5):", x);

x -= 3;
console.log("Subtraction Assignment (x -= 3):", x);

x *= 2;
console.log("Multiplication Assignment (x *= 2):", x);

x /= 4;
console.log("Division Assignment (x /= 4):", x);
